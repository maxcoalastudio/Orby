<?php
class Usuarios{
    private $pdo;
    public function __construct($dbname, $host, $user, $senha){
        try{
            $this->pdo = new PDO("mysql:dbname=".$dbname.";host=".$host, $user, $senha);
            $this->pdo->setAttribute(PDO::ATTR_ERRMODE,PDO::ERRMODE_EXCEPTION);
        }catch(PDOException $e){
            echo "Erro ao se conectar ao banco de dados: " .$e->getMessage();
            exit();
        }catch(Exception $e){
            echo "Erro generico: ". $e->getMessage();
            exit();
        }
    }

    public function criandoUsuario($email, $usuario, $senha){
        //Verifica email se existe antes de cadastrar
        $sql = "SELECT id FROM usuarios WHERE email = :email";
        $stmt = $this->pdo->prepare($sql);
        $stmt->bindValue(":email", $email);
        $stmt->execute();
        if($stmt->rowCount() > 0){
            $_SESSION['erro'] = "Usuário já cadastrado!";
            header('Location: login.php');
            exit();
        }else{//se não existir esse email no registro
            $sql = "INSERT INTO usuarios(email, usuario, senha) VALUES(:email, :usuario, :senha )";
            $stmt = $this->pdo->prepare($sql);
            $stmt->bindValue(":email", $email);
            $stmt->bindValue(":usuario", $usuario);
            $stmt->bindValue(":senha", $senha);
            if ($stmt->execute()){
                $_SESSION['sucesso'] = "Usuário criado com sucesso!";
                header('Location: login.php');
                exit();
            } else {
                $_SESSION['erro'] = "Erro ao criar usuário!";
                header('Location: login.php');
                exit();
            }
        }
    }

    public function buscarUsuario($email, $usuario, $senha){
        $res = array();
        $sql = "SELECT * FROM usuarios WHERE email = :email AND usuario = :usuario LIMIT 1";
        $stmt = $this->pdo->prepare($sql);
        $stmt->bindValue(":email", $email);
        $stmt->bindValue(":usuario", $usuario);
        if ($stmt->execute()){
            if($stmt->rowCount() > 0){
                $res = $stmt->fetch(PDO::FETCH_ASSOC);
                //vamos verificar a senha bate com o dados
                $senha_hash = $res['senha'];
                if(password_verify($senha, $senha_hash)){
                    //se a senha for essa mesma vamos criar uma sessao para isso
                    $_SESSION['usuario_id'] = $res['id'];
                    //vamos armazenar o id e vamos criar outra para nome
                    $_SESSION['usuario_nome'] = $res['usuario'];
                    //feito isso e deu certinho vamos entrar em uma area restrita
                    header('Location: restrita.php');
                    exit();
                }else{
                    //se ele não logar vamos fazer ele apresentar um erro, vamos criar quele login de erro de sessão que ja colocamos antes
                    $_SESSION['erro'] = "Senha Incorreta!";
                    //agora vamos carregar outra pagina
                    header('Location: login.php');
                    exit();
                }//agora se ele não encontrar usuario
            }else{
                $_SESSION['erro'] = "Usuario não encontrado!";
                header('Location: login.php');
                exit();
            }
        } else {
            echo "Erro ao executar a consulta: " . $stmt->errorInfo()[2];
        }
    }
}
